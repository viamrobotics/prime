# Code Editor

Uses `monaco` for syntax-highlighting.

<Canvas>
  <Story
    name="JSON editor"
    args={{
      language: "json",
      value: `{ 
  "foo": "bar",
  "some": "thing",
  "yes": true,
  "count": 10
}`,
    }}
  >
    {({ value, language, theme, readonly, minimap, uri }) => `
    <div style='width: 400px; height: 400px;'>
      <v-code-editor 
        value='${value}'
        language='${language}'
        theme='${theme}'
        readonly='${readonly}'
        minimap='${minimap}'
        uri='${uri ?? ''}'
      />
    </div>
    `}
  </Story>
</Canvas>
